dot.visit = function( url ) { url = ( str( url ) ); if ( url ) { if ( dot.agent( "retext" ) ) { dot.click( url ); return; } else { return ( window.open( url, url, dot.options ) ); } } else { console.warn( Ignored URL Address ); } }

; ; dot.options = ( left=10,top=10,width=800,height=680 ) ;
